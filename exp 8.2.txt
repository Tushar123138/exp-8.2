<script>
let token = null;

// Simulate backend login
function login() {
    const user = document.getElementById("username").value;
    const pass = document.getElementById("password").value;
    const output = document.getElementById("output");

    if (user === "testuser" && pass === "password123") {
        token = btoa(JSON.stringify({
            id: 1,
            username: user,
            iat: Date.now(),
            exp: Date.now() + 60000 // expires in 1 minute
        }));

        output.textContent = JSON.stringify({
            message: "Login success",
            token: token
        }, null, 2);
    } else {
        output.textContent = JSON.stringify({
            message: "Invalid credentials"
        }, null, 2);
    }
}

// Protected route simulation
function accessProtected() {
    const output = document.getElementById("output");

    if (!token) {
        output.textContent = JSON.stringify({ message: "Token missing" }, null, 2);
        return;
    }

    // decode token
    let decoded = JSON.parse(atob(token));
    const now = Date.now();

    if (decoded.exp < now) {
        output.textContent = JSON.stringify({ message: "Invalid or expired token" }, null, 2);
    } else {
        output.textContent = JSON.stringify({
            message: "âœ… Protected route accessed",
            user: {
                id: decoded.id,
                username: decoded.username,
                iat: decoded.iat,
                exp: decoded.exp
            }
        }, null, 2);
    }
}
